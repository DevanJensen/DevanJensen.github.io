Organizing Code in JavaScript

If it made your eyes soar to look that the nested if statement in the previous page, then I have good news for you.

You could re-organize that code and use functions to potentially make is easier follow.

Here's the code that we worked with previously... 

var likesBooks = prompt("Do you like books (enter 'yes' or 'no')?");

if(likesBooks == "yes"){
  // the user likes books...

  var likesSciFi = prompt("Do you like Sci Fi books (enter 'yes' or 'no')?");
  
  if(likesSciFi == "yes"){
    alert("I recommend Frankenstein!");
  }else{
    alert("I recommend Gone With the Wind!");
  }

}else if(likesBooks == "no"){
  // the user does NOT like books...

  var likesMovies = prompt("Do you like movies (enter 'yes' or 'no')?");
  
  if(likesMovies == "yes"){
    alert("I recommend Social Networking!");
  }else{
    alert("I recommend knitting!");
  }

}else{

  alert("It was a yes/no question! And remember that JS is case sensitive!");

}

Here's an alternative approach that you could use, which you may find easier to follow. We could create functions that handle the nested if statements like this: 

function checkSciFiBooks(){

  var likesSciFi = prompt("Do you like Sci Fi books (enter 'yes' or 'no')?");
  
  if(likesSciFi == "yes"){
    alert("I recommend Frankenstein!");
  }else{
    alert("I recommend Gone With the Wind!");
  }

}

function checkMovies(){

  var likesMovies = prompt("Do you like movies (enter 'yes' or 'no')?");
  
  if(likesMovies == "yes"){
    alert("I recommend Social Networking!");
  }else{
    alert("I recommend knitting!");
  }

}

Then our program might look like this: 

var likesBooks = prompt("Do you like books (enter 'yes' or 'no')?");

if(likesBooks == "yes"){
  checkSciFiBooks();
}else if(likesBooks == "no"){
  checkMovies();
}else{
  alert("It was a yes/no question! And remember that JS is case sensitive!");
}
